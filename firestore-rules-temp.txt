# TEMPORARY FIRESTORE RULES FOR TESTING
# Copy and paste these rules in Firebase Console > Firestore > Rules
# DO NOT leave these rules in production - they allow any signed-in user to read everything

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      allow read: if request.auth != null;   // any signed-in user can read
      allow write: if false;
    }
  }
}
