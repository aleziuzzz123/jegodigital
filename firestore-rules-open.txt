# OPEN FIRESTORE RULES FOR TESTING (NO AUTH REQUIRED)
# Copy and paste these rules in Firebase Console > Firestore > Rules
# DO NOT leave these rules in production - they allow anyone to read everything

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /{document=**} {
      allow read: if true;   // anyone can read
      allow write: if false; // no one can write
    }
  }
}
